# CIFAR-10 main experiments for ASAGuard
# Run from the project root directory, e.g.: cd ~/projects/ASAGuard
# These commands use physical GPU 1 via CUDA_VISIBLE_DEVICES=1.
# Each command runs in the background and overwrites its log under logs/cifar10/.
# If GPU memory is limited, run one command at a time instead of launching the whole block.

mkdir -p logs/cifar10

# ===== Attack: model_replacement =====
CUDA_VISIBLE_DEVICES=1 nohup python scripts/run_main_experiments.py --dataset cifar10 --attack model_replacement --defense fedavg --run > logs/cifar10/model_replacement_fedavg_seed123.log 2>&1 &
CUDA_VISIBLE_DEVICES=1 nohup python scripts/run_main_experiments.py --dataset cifar10 --attack model_replacement --defense asaguard --run > logs/cifar10/model_replacement_asaguard_seed123.log 2>&1 &
CUDA_VISIBLE_DEVICES=1 nohup python scripts/run_main_experiments.py --dataset cifar10 --attack model_replacement --defense multi_krum --run > logs/cifar10/model_replacement_multi_krum_seed123.log 2>&1 &
CUDA_VISIBLE_DEVICES=1 nohup python scripts/run_main_experiments.py --dataset cifar10 --attack model_replacement --defense trimmed_mean --run > logs/cifar10/model_replacement_trimmed_mean_seed123.log 2>&1 &
CUDA_VISIBLE_DEVICES=1 nohup python scripts/run_main_experiments.py --dataset cifar10 --attack model_replacement --defense flame --run > logs/cifar10/model_replacement_flame_seed123.log 2>&1 &
CUDA_VISIBLE_DEVICES=1 nohup python scripts/run_main_experiments.py --dataset cifar10 --attack model_replacement --defense flip --run > logs/cifar10/model_replacement_flip_seed123.log 2>&1 &
CUDA_VISIBLE_DEVICES=1 nohup python scripts/run_main_experiments.py --dataset cifar10 --attack model_replacement --defense fdcr --run > logs/cifar10/model_replacement_fdcr_seed123.log 2>&1 &
